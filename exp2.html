To implement the Minimax algorithm with Alpha-Beta pruning for the Tic-Tac-Toe game

AIM

To implement the Minimax algorithm for the Tic-Tac-Toe game using Python in order to determine the optimal move for the AI player.

ALGORITHM

  1. Initialize a 3×3 Tic-Tac-Toe board with empty positions.
  2. Define all possible winning combinations for the game board.
  3. Check whether a player has won or if the board is full to identify terminal states.
  4. Implement the Minimax function to evaluate board positions recursively.
  5. Assign scores to terminal states: +1 for AI win, −1 for human win, and 0 for draw.
  6. Maximize the score for the AI player and minimize the score for the human player.
  7. Select the move that yields the best Minimax score for the AI.
  8. Repeat the process until the game ends with a win or a draw.

SOURCE CODE

board = [' ' for _ in range(9)]
def print_board():
    print()
    for i in range(3):
        print("|".join(board[i*3:(i+1)*3]))
        if i < 2:
            print(" ------- ")
    print()
def check_winner(b):
    wins = [(0,1,2),(3,4,5),(6,7,8),
            (0,3,6),(1,4,7),(2,5,8),
            (0,4,8),(2,4,6)]
    for a, b1, c in wins:
        if b[a] == b[b1] == b[c] and b[a] != ' ':
            return b[a]
    return None
def board_full(b):
    return ' ' not in b
def minimax(b, is_maximizing):
    winner = check_winner(b)
    if winner == 'X':
        return 1
    elif winner == 'O':
        return -1
    elif board_full(b):
        return 0
    if is_maximizing:
        best_score = -100
        for i in range(9):
            if b[i] == ' ':
                b[i] = 'X'
                score = minimax(b, False)
                b[i] = ' '
                best_score = max(score, best_score)
        return best_score
    else:
        best_score = 100
        for i in range(9):
            if b[i] == ' ':
                b[i] = 'O'
                score = minimax(b, True)
                b[i] = ' '
                best_score = min( score, best_score)
        return best_score
def ai_move():
    best_score = -100
    move = 0
    for i in range(9):
        if board[i] == ' ':
            board[i] = 'X'
            score = minimax(board, False)
            board[i] = ' '
            if score > best_score:
                best_score = score
                move = i
    return move
print("TIC-TAC-TOE using MINIMAX")
print("AI = X | Human = O")
print("Positions:")
print("0 1 2 \n3 4 5 \n6 7 8")
print_board()
while True:
    move = ai_move()
    board[move] = 'X'
    print("AI played:", move)
    print_board()
    if check_winner(board):
        print("Winner:", check_winner(board))
        break
    if board_full(board):
        print("Game Draw")
        break
    pos = int(input("Your move (0-8): "))
    if board[pos] != ' ':
        print("Invalid move. Try again.")
        continue
    board[pos] = 'O'
    print_board()
    if check_winner(board):
        print("Winner:", check_winner(board))
        break
    if board_full(board):
        print("Game Draw")
        break
